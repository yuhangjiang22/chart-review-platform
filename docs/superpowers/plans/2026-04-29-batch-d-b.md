# Batch D-B Implementation Plan (protocol version graph + migration UI)

> **For agentic workers:** REQUIRED SUB-SKILL: superpowers:subagent-driven-development. Steps use checkbox (`- [ ]`) syntax.

**Goal:** Ship the version graph + migration tooling so locked records can be diffed across versions, an impact simulator predicts which records change under a new version, and a targeted re-run reopens only the affected records. Closes Beat 12 (☐ → ✓).

**Architecture:** Lock-time hook archives compiled task JSON to `tasks/<tid>/versions/<sha>.json`. Three new server modules (version-archive, task-diff, impact-simulator, migration). Heuristic impact: records are "affected" if they answered any field whose `is_applicable_when`/`derivation`/`answer_schema`/`guidance_prose` changed in the diff. Targeted re-run archives the locked record to `_archive/<from_sha>.json` and reopens to `agent_complete`. Methodologist gains a Revision History section; Studio gains a MigrationPanel.

**Tech Stack:** TypeScript 5.5, Express 4.21, vitest. No new deps.

**Spec:** `docs/superpowers/specs/2026-04-29-batch-d-b-design.md`.

**Effort:** ~7-10 days. 14 tasks across 6 phases.

```ts
// app/server/version-archive.ts
import fs from "fs";
import path from "path";

function platformRoot(): string {
  return process.env.CHART_REVIEW_PLATFORM_ROOT ?? path.resolve(__dirname, "../..");
}

export interface VersionEntry {
  task_id: string;
  lock_task_sha: string;
  archived_at: string;
  record_count: number;
  task_version?: string;
}

function versionsDir(taskId: string): string {
  return path.join(platformRoot(), "tasks", taskId, "versions");
}

function compiledTaskPath(taskId: string): string {
  return path.join(platformRoot(), "tasks", "compiled", `${taskId}.json`);
}

export function archiveVersion(taskId: string, lockTaskSha: string): void {
  const target = path.join(versionsDir(taskId), `${lockTaskSha}.json`);
  if (fs.existsSync(target)) return;  // idempotent
  const source = compiledTaskPath(taskId);
  if (!fs.existsSync(source)) return;  // no compiled task to snapshot
  fs.mkdirSync(versionsDir(taskId), { recursive: true });
  fs.copyFileSync(source, target);
}

export function listVersions(taskId: string, reviewsRoot: string): VersionEntry[] {
  const dir = versionsDir(taskId);
  if (!fs.existsSync(dir)) return [];

  const out: VersionEntry[] = [];
  for (const f of fs.readdirSync(dir)) {
    if (!f.endsWith(".json")) continue;
    const sha = f.replace(/\.json$/, "");
    const stat = fs.statSync(path.join(dir, f));
    let task_version: string | undefined;
    try {
      const data = JSON.parse(fs.readFileSync(path.join(dir, f), "utf8")) as { task_version?: string };
      task_version = data.task_version;
    } catch { /* skip */ }
    out.push({
      task_id: taskId,
      lock_task_sha: sha,
      archived_at: stat.mtime.toISOString(),
      record_count: countRecordsForSha(taskId, sha, reviewsRoot),
      task_version,
    });
  }
  return out.sort((a, b) => (a.archived_at < b.archived_at ? 1 : -1));
}

export function loadVersionedTask(taskId: string, lockTaskSha: string): unknown {
  const p = path.join(versionsDir(taskId), `${lockTaskSha}.json`);
  if (!fs.existsSync(p)) return null;
  try { return JSON.parse(fs.readFileSync(p, "utf8")); } catch { return null; }
}

function countRecordsForSha(taskId: string, sha: string, reviewsRoot: string): number {
  if (!fs.existsSync(reviewsRoot)) return 0;
  let n = 0;
  for (const pid of fs.readdirSync(reviewsRoot)) {
    if (pid.startsWith("_")) continue;
    const rsPath = path.join(reviewsRoot, pid, taskId, "review_state.json");
    if (!fs.existsSync(rsPath)) continue;
    try {
      const rs = JSON.parse(fs.readFileSync(rsPath, "utf8")) as { lock_task_sha?: string };
      if (rs.lock_task_sha === sha) n++;
    } catch {}
  }
  return n;
}
```

```ts
// app/server/task-diff.ts
export interface FieldDiff {
  field_id: string;
  status: "added" | "removed" | "changed" | "unchanged";
  changes?: Array<{ key: string; from: unknown; to: unknown }>;
}

export interface TaskDiff {
  from_sha: string;
  to_sha: string;
  fields: FieldDiff[];
  global_changes: Array<{ key: string; from: unknown; to: unknown }>;
}

const FIELD_KEYS_TO_DIFF = [
  "prompt",
  "guidance_prose",
  "answer_schema",
  "is_applicable_when",
  "derivation",
  "is_final_output",
  "requires_calibration",
];

const TASK_GLOBAL_KEYS_TO_DIFF = [
  "source_document_sha",
  "task_version",
  "review_unit",
  "stratify_by",
];

interface TaskShape {
  fields?: Array<Record<string, unknown> & { id: string }>;
  [k: string]: unknown;
}

export function computeTaskDiff(fromTask: unknown, toTask: unknown, fromSha = "", toSha = ""): TaskDiff {
  const from = (fromTask ?? {}) as TaskShape;
  const to = (toTask ?? {}) as TaskShape;

  // Field-level diff
  const fromFields = new Map((from.fields ?? []).map((f) => [f.id, f]));
  const toFields = new Map((to.fields ?? []).map((f) => [f.id, f]));
  const allIds = new Set([...fromFields.keys(), ...toFields.keys()]);

  const fields: FieldDiff[] = [];
  for (const id of allIds) {
    const f = fromFields.get(id);
    const t = toFields.get(id);
    if (!f) {
      fields.push({ field_id: id, status: "added" });
    } else if (!t) {
      fields.push({ field_id: id, status: "removed" });
    } else {
      const changes: Array<{ key: string; from: unknown; to: unknown }> = [];
      for (const k of FIELD_KEYS_TO_DIFF) {
        const a = JSON.stringify(f[k]);
        const b = JSON.stringify(t[k]);
        if (a !== b) changes.push({ key: k, from: f[k], to: t[k] });
      }
      fields.push({
        field_id: id,
        status: changes.length === 0 ? "unchanged" : "changed",
        ...(changes.length > 0 && { changes }),
      });
    }
  }

  // Global-level diff
  const global_changes: Array<{ key: string; from: unknown; to: unknown }> = [];
  for (const k of TASK_GLOBAL_KEYS_TO_DIFF) {
    const a = JSON.stringify(from[k]);
    const b = JSON.stringify(to[k]);
    if (a !== b) global_changes.push({ key: k, from: from[k], to: to[k] });
  }

  return { from_sha: fromSha, to_sha: toSha, fields, global_changes };
}
```

```ts
// app/server/impact-simulator.ts
import fs from "fs";
import path from "path";
import { computeTaskDiff } from "./task-diff.js";
import { loadVersionedTask } from "./version-archive.js";

export interface ImpactInput {
  taskId: string;
  fromSha: string;
  toSha: string;
  reviewsRoot: string;
}

export interface ImpactResult {
  total_locked: number;
  total_unlocked: number;
  affected: Array<{ patient_id: string; review_status: string; intersect_fields: string[] }>;
  unaffected: string[];
  changed_field_ids: string[];
}

const STRUCTURAL_KEYS = new Set([
  "is_applicable_when",
  "derivation",
  "answer_schema",
  "guidance_prose",
  "prompt",
]);

export function simulateImpact(input: ImpactInput): ImpactResult {
  const { taskId, fromSha, toSha, reviewsRoot } = input;
  const fromTask = loadVersionedTask(taskId, fromSha);
  const toTask = loadVersionedTask(taskId, toSha);
  const diff = computeTaskDiff(fromTask, toTask, fromSha, toSha);

  // Build the structurally-changed field set
  const changed = new Set<string>();
  for (const f of diff.fields) {
    if (f.status === "added" || f.status === "removed") {
      changed.add(f.field_id);
    } else if (f.status === "changed") {
      const hasStructural = (f.changes ?? []).some((c) => STRUCTURAL_KEYS.has(c.key));
      if (hasStructural) changed.add(f.field_id);
    }
  }

  // Walk locked records under fromSha
  const affected: Array<{ patient_id: string; review_status: string; intersect_fields: string[] }> = [];
  const unaffected: string[] = [];
  let totalLocked = 0;
  let totalUnlocked = 0;

  if (fs.existsSync(reviewsRoot)) {
    for (const pid of fs.readdirSync(reviewsRoot)) {
      if (pid.startsWith("_")) continue;
      const rsPath = path.join(reviewsRoot, pid, taskId, "review_state.json");
      if (!fs.existsSync(rsPath)) continue;
      try {
        const rs = JSON.parse(fs.readFileSync(rsPath, "utf8")) as {
          review_status?: string;
          lock_task_sha?: string;
          field_assessments?: Array<{ field_id: string }>;
        };
        if (rs.review_status === "locked" && rs.lock_task_sha === fromSha) {
          totalLocked++;
          const recordFields = new Set((rs.field_assessments ?? []).map((fa) => fa.field_id));
          const intersect = [...changed].filter((f) => recordFields.has(f));
          if (intersect.length > 0) {
            affected.push({ patient_id: pid, review_status: rs.review_status, intersect_fields: intersect });
          } else {
            unaffected.push(pid);
          }
        } else {
          totalUnlocked++;
        }
      } catch {
        // skip
      }
    }
  }

  return {
    total_locked: totalLocked,
    total_unlocked: totalUnlocked,
    affected,
    unaffected,
    changed_field_ids: [...changed],
  };
}
```

```ts
// app/server/migration.ts
import fs from "fs";
import path from "path";
import { appendAuditEntry } from "./audit-trail.js";

export interface MigrationInput {
  taskId: string;
  fromSha: string;
  toSha: string;
  patientIds: string[];
  reviewsRoot: string;
  triggeredBy: string;
}

export interface MigrationResult {
  archived: string[];
  reopened: string[];
  errors: Array<{ patient_id: string; error: string }>;
}

export async function runMigration(input: MigrationInput): Promise<MigrationResult> {
  const { taskId, fromSha, toSha, patientIds, reviewsRoot, triggeredBy } = input;
  const archived: string[] = [];
  const reopened: string[] = [];
  const errors: Array<{ patient_id: string; error: string }> = [];

  for (const pid of patientIds) {
    const rsPath = path.join(reviewsRoot, pid, taskId, "review_state.json");
    if (!fs.existsSync(rsPath)) {
      errors.push({ patient_id: pid, error: "review_state not found" });
      continue;
    }
    let rs: Record<string, unknown>;
    try {
      rs = JSON.parse(fs.readFileSync(rsPath, "utf8"));
    } catch (e) {
      errors.push({ patient_id: pid, error: `parse error: ${e}` });
      continue;
    }

    if (rs.review_status !== "locked" || rs.lock_task_sha !== fromSha) {
      errors.push({ patient_id: pid, error: `record not locked under ${fromSha} (status=${rs.review_status}, sha=${rs.lock_task_sha})` });
      continue;
    }

    // Archive the current state
    const archiveDir = path.join(reviewsRoot, pid, taskId, "_archive");
    fs.mkdirSync(archiveDir, { recursive: true });
    const archivePath = path.join(archiveDir, `${fromSha}.json`);
    fs.writeFileSync(archivePath, JSON.stringify(rs, null, 2));

    // Reopen the record (clear lock fields, set agent_complete)
    const reopened_rs = {
      ...rs,
      review_status: "agent_complete",
      version: ((rs.version as number) ?? 0) + 1,
      updated_at: new Date().toISOString(),
      updated_by: triggeredBy,
    };
    delete (reopened_rs as Record<string, unknown>).lock_task_sha;
    delete (reopened_rs as Record<string, unknown>).locked_at;
    delete (reopened_rs as Record<string, unknown>).locked_by;
    fs.writeFileSync(rsPath, JSON.stringify(reopened_rs, null, 2));

    // Audit
    const ts = new Date().toISOString();
    const sessionId = `migrate-${Date.now()}`;
    appendAuditEntry(
      { patientId: pid, taskId, sessionId },
      {
        ts, session_id: sessionId,
        step_type: "record_superseded",
        from_sha: fromSha,
        to_sha: toSha,
        archived_path: `_archive/${fromSha}.json`,
        triggered_by: triggeredBy,
      } as never,
    );

    archived.push(pid);
    reopened.push(pid);
  }

  // One global migration_run audit (use task-level sessionId)
  if (archived.length > 0) {
    const ts = new Date().toISOString();
    const sessionId = `migrate-${Date.now()}`;
    // Pick any patient (first archived) as the audit-coordinate patient — audit JSONL is per-(pid, tid) so this can't be truly task-scoped.
    // Choose to also write a task-scoped record to the first archived patient's audit log.
    appendAuditEntry(
      { patientId: archived[0], taskId, sessionId },
      {
        ts, session_id: sessionId,
        step_type: "migration_run",
        from_sha: fromSha,
        to_sha: toSha,
        affected_count: archived.length,
        triggered_by: triggeredBy,
      } as never,
    );
  }

  return { archived, reopened, errors };
}
```

```python
    if not r.ok or len(r.json()) < 2:

    from_sha = versions[1]["lock_task_sha"]
    to_sha = versions[0]["lock_task_sha"]
    # Simulate
```
