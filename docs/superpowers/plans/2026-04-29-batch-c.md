```ts
// app/server/kappa.ts
import fs from "fs";
import path from "path";

export interface ReplayedAnswer {
  patient_id: string;
  reviewer_id: string;
  field_id: string;
  answer: unknown;
  ts: string;
}

export interface KappaResult {
  kappa: number;
  kappa_reviewers: [string, string];
  kappa_n_shared: number;
  confusion: Record<string, Record<string, number>>;
}

/** Walk all chat/<*>.jsonl files for a (task, field) pair; reconstruct
 *  (pid, reviewer, fid, answer, ts) tuples from ui_action entries with
 *  set_field_assessment. Last-write-wins per (patient, reviewer, field). */
export function replayReviewerAnswers(reviewsRoot: string, taskId: string, fieldId: string): ReplayedAnswer[] {
  const latest = new Map<string, ReplayedAnswer>();  // key: pid|reviewer
  if (!fs.existsSync(reviewsRoot)) return [];

  for (const pid of fs.readdirSync(reviewsRoot)) {
    if (pid.startsWith("_")) continue;
    const chatDir = path.join(reviewsRoot, pid, taskId, "chat");
    if (!fs.existsSync(chatDir)) continue;
    for (const f of fs.readdirSync(chatDir)) {
      const lines = fs.readFileSync(path.join(chatDir, f), "utf8").split("\n").filter(Boolean);
      for (const line of lines) {
        try {
          const entry = JSON.parse(line) as {
            step_type?: string;
            action_type?: string;
            source?: string;
            payload_field_id?: string;
            payload_answer?: unknown;
            ts?: string;
          };
          if (
            entry.step_type !== "ui_action" ||
            entry.action_type !== "set_field_assessment" ||
            entry.source !== "reviewer" ||
            entry.payload_field_id !== fieldId ||
            !entry.ts
          ) continue;
          // Need to know the reviewer_id — read the parent state file's updated_by
          // OR read added_evidence_id / similar. Actually the payload_summary or
          // the audit entry's session_id encodes it. The cleanest place is to read
          // the entry's `result_version`-corresponding state at that ts.
          // But for v1, we use the heuristic: each session_id is owned by ONE
          // reviewer (auth-bound), so the latest state file's updated_by at the
          // session start gives us the reviewer. Simpler: extract from the
          // payload_summary if present.
          const summary = (entry as { payload_summary?: string }).payload_summary ?? "";
          const reviewerMatch = summary.match(/reviewer=([\w@.-]+)/);
          const reviewerFromSummary = reviewerMatch ? reviewerMatch[1] : null;
          // Fallback: walk the per-patient review_state.json's audit context
          // For v1 simplicity, attempt to read updated_by from the latest state
          // OR extract from session_id pattern (e.g., "reviewer__alice").
          const sessionMatch = ((entry as { session_id?: string }).session_id ?? "").match(/^reviewer__([^_]+)$/);
          const reviewerId = reviewerFromSummary ?? (sessionMatch ? sessionMatch[1] : null);
          if (!reviewerId) continue;  // can't disambiguate

          const key = `${pid}|${reviewerId}`;
          const existing = latest.get(key);
          if (!existing || existing.ts < entry.ts) {
            latest.set(key, {
              patient_id: pid,
              reviewer_id: reviewerId,
              field_id: fieldId,
              answer: entry.payload_answer,
              ts: entry.ts,
            });
          }
        } catch {
          // skip malformed
        }
      }
    }
  }
  return [...latest.values()];
}

/** Compute Cohen's κ for the 2 most-frequent reviewers in the replayed answers,
 *  restricted to shared records (both reviewers wrote on the same patient).
 *  Returns null if fewer than 10 shared records. */
export function computeKappaProper(answers: ReplayedAnswer[]): KappaResult | null {
  // Identify the 2 most-frequent reviewers
  const byReviewer = new Map<string, ReplayedAnswer[]>();
  for (const a of answers) {
    const arr = byReviewer.get(a.reviewer_id) ?? [];
    arr.push(a);
    byReviewer.set(a.reviewer_id, arr);
  }
  const reviewers = [...byReviewer.entries()]
    .sort((a, b) => b[1].length - a[1].length)
    .slice(0, 2)
    .map((e) => e[0]);
  if (reviewers.length < 2) return null;
  const [rA, rB] = reviewers as [string, string];

  // Pair up by patient_id
  const aByPid = new Map(byReviewer.get(rA)!.map((a) => [a.patient_id, a.answer]));
  const bByPid = new Map(byReviewer.get(rB)!.map((a) => [a.patient_id, a.answer]));
  const sharedPids = [...aByPid.keys()].filter((p) => bByPid.has(p));
  if (sharedPids.length < 10) return null;

  // Compute κ
  const total = sharedPids.length;
  let agreements = 0;
  const confusion: Record<string, Record<string, number>> = {};
  const aCounts: Record<string, number> = {};
  const bCounts: Record<string, number> = {};
  for (const pid of sharedPids) {
    const a = String(aByPid.get(pid));
    const b = String(bByPid.get(pid));
    if (a === b) agreements++;
    if (!confusion[a]) confusion[a] = {};
    confusion[a][b] = (confusion[a][b] ?? 0) + 1;
    aCounts[a] = (aCounts[a] ?? 0) + 1;
    bCounts[b] = (bCounts[b] ?? 0) + 1;
  }

  const Po = agreements / total;
  const categories = new Set([...Object.keys(aCounts), ...Object.keys(bCounts)]);
  let Pe = 0;
  for (const c of categories) {
    Pe += ((aCounts[c] ?? 0) / total) * ((bCounts[c] ?? 0) / total);
  }

  const kappa = 1 - Pe === 0 ? 1.0 : (Po - Pe) / (1 - Pe);

  return {
    kappa,
    kappa_reviewers: [rA, rB],
    kappa_n_shared: total,
    confusion,
  };
}
```

**Note on reviewer disambiguation**: the implementation extracts reviewer_id from the audit entry's `session_id` (the format `reviewer__<name>` was established in Phase B Task 23). Older entries without that pattern are skipped. If your test seeding uses a different session_id format, adjust the regex OR seed session_ids matching `reviewer__alice` etc.

```ts
// app/server/methodologist-pdf.ts
import PDFDocument from "pdfkit";
import { Readable } from "stream";
import { computeQAStats, type QAStats } from "./qa-panel.js";
import { loadCompiledTask } from "./tasks.js";

export async function generatePdf(taskId: string, reviewsRoot: string): Promise<Readable> {
  const task = await loadCompiledTask(taskId);
  const qa = await computeQAStats(taskId, reviewsRoot);

  const doc = new PDFDocument({ size: "letter", margin: 50 });

  // Page 1: Header
  doc.fontSize(20).text(`Chart Review Verification Report`, { align: "center" });
  doc.moveDown(0.5);
  doc.fontSize(14).text(`Task: ${taskId}`, { align: "center" });
  doc.moveDown();
  doc.fontSize(10).text(`Generated: ${new Date().toISOString()}`);
  doc.moveDown();
  doc.text(`Total records: ${qa.total_records}`);
  doc.text(`Locked: ${qa.records_locked}`);
  doc.text(`Validated: ${qa.records_validated}`);
  doc.text(`In progress: ${qa.records_in_progress}`);

  // Page 2: Locked task contract
  doc.addPage();
  doc.fontSize(16).text("Task contract");
  doc.moveDown();
  if (task && (task as { fields?: Array<{ id: string; prompt?: string }> }).fields) {
    for (const f of (task as { fields: Array<{ id: string; prompt?: string }> }).fields) {
      doc.fontSize(12).font("Helvetica-Bold").text(f.id);
      if (f.prompt) {
        doc.font("Helvetica").fontSize(10).text(f.prompt, { paragraphGap: 6 });
      }
      doc.moveDown(0.5);
    }
  }

  // Page 3+: Calibration metrics
  doc.addPage();
  doc.fontSize(16).text("Calibration metrics");
  doc.moveDown();
  for (const [fid, c] of Object.entries(qa.by_criterion)) {
    doc.fontSize(12).font("Helvetica-Bold").text(fid);
    doc.font("Helvetica").fontSize(10);
    doc.text(`Override rate: ${(c.override_rate * 100).toFixed(1)}% (${c.override_count}/${c.reviewer_touched})`);
    if (c.kappa !== undefined) {
      doc.text(`κ = ${c.kappa.toFixed(3)} (${c.kappa_reviewers?.[0]} vs ${c.kappa_reviewers?.[1]}, ${c.kappa_n_shared} shared)`);
    }
    if (Object.keys(c.override_reasons).length > 0) {
      doc.text(`Reasons: ${Object.entries(c.override_reasons).map(([r, n]) => `${r}: ${n}`).join(", ")}`);
    }
    doc.moveDown();
  }

  // Drift alerts
  if (qa.drift_alerts.length > 0) {
    doc.addPage();
    doc.fontSize(16).text("Drift alerts");
    doc.moveDown();
    for (const alert of qa.drift_alerts) {
      doc.fontSize(10);
      doc.text(`${alert.field_id}: baseline ${(alert.baseline_rate*100).toFixed(1)}% → current ${(alert.current_rate*100).toFixed(1)}% (Δ${alert.delta_pp.toFixed(1)}pp), triggered ${alert.triggered_at.slice(0, 19)}`);
      doc.moveDown(0.3);
    }
  }

  doc.end();
  return doc as unknown as Readable;
}
```

```ts
// app/server/methods-drafter.ts
import { query } from "@anthropic-ai/claude-agent-sdk";
import { computeQAStats } from "./qa-panel.js";
import { loadCompiledTask } from "./tasks.js";

const SYSTEM_PROMPT = `You are drafting the methods section of an academic paper from this chart review study.

You will receive JSON with:
- task: the compiled task contract (criterion definitions, types, etc.)
- qa: calibration metrics (override rates, κ, drift, record counts)

Write a methods section (~300-500 words) following these conventions:
- Past tense, third person
- Cite the record count + reviewer count + κ if computed
- Describe the protocol structure: criterion definitions, derivations, gates
- Quote criterion definitions verbatim where they're load-bearing
- End with a paragraph on limitations (cohort scope, reviewer subjectivity, etc.)

Output ONLY the markdown methods section. No preamble, no commentary.`;

export async function draftMethodsSection(taskId: string, reviewsRoot: string): Promise<string> {
  const task = await loadCompiledTask(taskId);
  const qa = await computeQAStats(taskId, reviewsRoot);

  const userMessage = JSON.stringify({ task, qa }, null, 2);

  const stream = query({
    prompt: [
      { role: "user", content: [{ type: "text", text: userMessage }] }
    ] as never,
    options: {
      maxTurns: 1,
      model: process.env.CHART_REVIEW_MODEL ?? "anthropic/claude-haiku-4.5",
      systemPrompt: SYSTEM_PROMPT,
      mcpServers: {},
      permissionMode: "default" as never,
    } as never,
  });

  let out = "";
  for await (const msg of stream as AsyncIterable<{ type: string; text?: string; message?: { content?: Array<{ type: string; text?: string }> } }>) {
    if (msg.type === "assistant" && msg.message?.content) {
      for (const block of msg.message.content) {
        if (block.type === "text" && block.text) out += block.text;
      }
    }
  }
  return out.trim();
}
```

```ts
// app/server/auto-role-c.ts
import fs from "fs";
import path from "path";

const AUTO_ROLE_C_THRESHOLD = 3;
const AUTO_ROLE_C_WINDOW_MS = 24 * 60 * 60 * 1000;
const AUTO_ROLE_C_COOLDOWN_MS = 24 * 60 * 60 * 1000;

export interface AutoRoleCInput {
  taskId: string;
  reviewsRoot: string;
  fieldId: string;
}

export function shouldAutoRoleC(input: AutoRoleCInput): boolean {
  const { taskId, reviewsRoot, fieldId } = input;
  if (!fs.existsSync(reviewsRoot)) return false;

  const now = Date.now();
  let driftCount = 0;
  let cooldownActive = false;

  for (const pid of fs.readdirSync(reviewsRoot)) {
    if (pid.startsWith("_")) continue;
    const chatDir = path.join(reviewsRoot, pid, taskId, "chat");
    if (!fs.existsSync(chatDir)) continue;
    for (const f of fs.readdirSync(chatDir)) {
      const lines = fs.readFileSync(path.join(chatDir, f), "utf8").split("\n").filter(Boolean);
      for (const line of lines) {
        try {
          const entry = JSON.parse(line) as { step_type?: string; field_id?: string; ts?: string };
          if (entry.field_id !== fieldId || !entry.ts) continue;
          const tsMs = new Date(entry.ts).getTime();
          if (entry.step_type === "drift_alert" && now - tsMs <= AUTO_ROLE_C_WINDOW_MS) {
            driftCount++;
          } else if (entry.step_type === "role_c_auto_run" && now - tsMs <= AUTO_ROLE_C_COOLDOWN_MS) {
            cooldownActive = true;
          }
        } catch {
          // skip
        }
      }
    }
  }

  return driftCount >= AUTO_ROLE_C_THRESHOLD && !cooldownActive;
}

export async function fireAutoRoleC(input: AutoRoleCInput): Promise<void> {
  // Existing Role C entry point lives in feedback.ts. Adapt the call to whatever
  // function is exported there. Common shapes:
  //   await runRoleC(input.taskId)
  //   await analyzeCohort(input.taskId)
  // Read app/server/feedback.ts to find the right name.
  // For v1, this is a thin shim that imports + calls.
  const { runRoleC } = await import("./feedback.js");
  if (typeof runRoleC === "function") {
    await runRoleC(input.taskId);
  }
}
```
